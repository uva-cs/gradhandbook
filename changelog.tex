\section*{Changelog based on the last revision (August 2024)}

\subsection*{General Updates}
\begin{itemize}
  \item \textbf{Formatting:} The handbook was converted to \LaTeX{} for improved formatting and readability.
  \item \textbf{Changelog:} A changelog section was added to document updates.
\end{itemize}

\subsection*{Ph.D. Qualification Committee}
\begin{itemize}
  \item \textbf{Simplified Process:} The process for forming the Ph.D. Qualification Committee was simplified and rewritten for clarity.
\end{itemize}

\subsection*{Ph.D. Qualifying Examination}
\begin{itemize}
  \item \textbf{Expected Timing:} Added expected timing for completing the qualification proposal and presentation.
  \item \textbf{Phase Structure:} The depth component of the qualifying examination was restructured into two phases (proposal and presentation) with detailed steps for each.
  \item \textbf{Proposal Document:} Clarified requirements for the proposal document, including mandatory disclosure of related papers.
  \item \textbf{Reading List:} Updated guidelines for the reading list included in the proposal document.
  \item \textbf{Presentation:} Added requirements for the presentation, including the use of numbered slides and public announcement.
\end{itemize}

\subsection*{Ph.D. Student Assessment Policy and Process}
\begin{itemize}
  \item \textbf{Assessment Frequency:} Specified that student progress will be evaluated twice a year, at the end of the Spring and Fall terms.
  \item \textbf{Confidential Information:} Provided a link for students to submit confidential information that will not be shared with their advisor.
\end{itemize}

\subsection*{First-Year Ph.D. Rotation Program}
\begin{itemize}
  \item \textbf{Eligibility:} Clarified eligibility criteria for the rotation program.
  \item \textbf{Procedure:} Detailed the procedure for each rotation, including proposal, execution, and report stages.
  \item \textbf{Outcomes:} Defined possible outcomes after each rotation and responsibilities of rotation advisors.
\end{itemize}

\subsection*{Ph.D. Proposal}
\begin{itemize}
  \item \textbf{Expected Timing:} Added expected timing for completing the Ph.D. proposal.
  \item \textbf{Proposal Document:} Clarified structure and page limit for the proposal document.
  \item \textbf{Proposal Presentation:} Added requirements for the proposal presentation, including public announcement and use of numbered slides.
\end{itemize}

\subsection*{Ph.D. Defense}
\begin{itemize}
  \item \textbf{Defense Form:} Added requirement for initiating the Ph.D. Dissertation Defense form on the morning of the defense.
  \item \textbf{Defense Presentation:} Clarified components of the defense presentation and requirements for distributing presentation materials.
\end{itemize}

\subsection*{Ph.D. Program Termination}
\begin{itemize}
  \item \textbf{Termination Process:} Detailed the process for terminating a Ph.D. program due to unsatisfactory performance, including notification and petition procedures.
\end{itemize}

\subsection*{Miscellaneous}
\begin{itemize}
  \item \textbf{Useful Forms and Resources:} Updated URLs and added new resources.
  \item \textbf{Policy on Graduate Student Leave:} Clarified expectations for GTAs and GRAs regarding absences and leave.
\end{itemize}
